select Employees.FirstName + ' ' + Employees.LastName as EmployeeName, 2024-year(Employees.BirthDate) as EmployeeAge, 2024-year(E.BirthDate) as ManagerAge from Employees join Employees as E on Employees.ReportsTo = E.EmployeeID where 2024-year(Employees.BirthDate) >= 60 and 2024-year(E.BirthDate) >= 70
